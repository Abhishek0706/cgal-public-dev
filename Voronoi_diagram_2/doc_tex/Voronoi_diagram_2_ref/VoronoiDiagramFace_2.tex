\begin{ccRefClass}[Voronoi_diagram_2<DG,AT,AP>::]{Face}

%% \ccHtmlCrossLink{}     %% add further rules for cross referencing links
%% \ccHtmlIndexC[concept]{} %% add further index entries
\ccDefinition

The class \ccRefName\ is the class provided by the
\ccc{Voronoi_diagram_2<DG,AT,AP>} class for Voronoi faces. Below we
present its interface.

\ccIsModel
\ccc{DefaultConstructible}, \ccc{CopyConstructible}, \ccc{Assignable},
\ccc{EqualityComparable}, \ccc{LessThanComparable}

\ccTypes
%
\ccTwo{Voronoi_diagram_2<DG,AT,AP>::Face::Halfedge_handle+}{}
\ccNestedType{Vertex}{A type for the vertices of the Voronoi diagram.}
\ccGlue
\ccNestedType{Halfedge}{A type for the halfedges of the Voronoi diagram.}
\ccGlue
\ccNestedType{Vertex_handle}{Handle for the vertices of the Voronoi diagram.}
\ccGlue
\ccNestedType{Face_handle}{Handle for the faces of the Voronoi diagram.}
\ccGlue
\ccNestedType{Halfedge_handle}{Handle for the halfedges of the Voronoi
  diagram.}
\ccGlue
\ccNestedType{Ccb_halfedge_circulator}{A type for a bidirectional
  circulator over the halfedges on the boundary of the face. The value
  type of the circulator is
  \ccc{CGAL::Voronoi_diagram_2<DG,AT,AP>::Halfedge}, and is convertible
  to \ccc{Halfedge_handle}.}
\ccGlue
\ccNestedType{Delaunay_graph}{A type for the Delaunay graph. It is a
  model of the \ccc{DelaunayGraph_2} concept.}
%
\ccGlue
\ccThree{typedef Delaunay_graph::Vertex_handle}{Delaunay_vertex_handle;}{}
\ccThreeToTwo
\ccTypedef{typedef Delaunay_graph::Vertex_handle
  Delaunay_vertex_handle;}{A type for the handle of the dual vertex.}

\ccCreationVariable{f}  %% choose variable name


\ccHeading{Access Methods}
\ccThree{Ccb_halfedge_circulator}{f.halfedge()+}{}
%
\ccMethod{Halfedge_handle halfedge();}{Returns an incident halfedge on
  the boundary of \ccc{f}.}
\ccGlue
\ccMethod{Ccb_halfedge_circulator ccb();}{Returns a bidirectional
  circulator for traversing the halfedges on the boundary of
  \ccc{f}. The halfedges are traversed in counterclockwise order.}
\ccGlue
\ccMethod{Delaunay_vertex_handle dual();}
{Returns a handle to the corresponding dual vertex in the Delaunay graph.}

\ccHeading{Predicate Methods}
\ccThree{bool}{f.is_halfedge_on_ccb(Halfedge e)+}{}
\ccMethod{bool is_unbounded();}{Returns \ccStyle{true} iff the face is
  an unbounded face in the Voronoi diagram.}
\ccMethod{bool is_halfedge_on_ccb(Halfedge e);}{Returns
  \ccStyle{true} iff \ccc{e} is a halfedge of the boundary of
  \ccc{f}.}
\ccGlue
\ccMethod{bool is_valid();}{Returns \ccStyle{true} iff the following
  conditions are met: the face is not rejected by the chosen
  adaptation policy;
  all its adjacent halfedges do not have zero length; all its adjacent
  halfedges return the face as their adjacent face.}

\end{ccRefClass}
