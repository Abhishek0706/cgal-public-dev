\begin{ccRefConcept}{AlphaShapeVertex_2}

\ccDefinition

\ccRefines
\ccc{TriangulationVertexBase_2}.



\ccCreationVariable{v}

\ccTypes

\ccNestedType{FT}{A type to hold a coordinate type class. 
The type must provide a copy constructor, assignment, comparison
operators, negation, multiplication, division and allow the
declaration and initialization with a small integer constant
(cf. requirements for number types). An obvious choice would be
coordinate type of the point class.}

\ccCreation

\begin{ccAdvanced}

\ccConstructor{AlphaShapeVertex_2();}{default constructor.}
\ccGlue
\ccConstructor{AlphaShapeVertex_2(Point p);}{constructor setting
the point.}
\ccGlue
\ccConstructor{AlphaShapeVertex_2(Point p, const Face_handle& ff);} 
{constructor setting the point  associated to and an incident face.}

\end{ccAdvanced}

\ccHeading{Access Functions}

\ccMethod{std::pair< FT, FT > get_range();}
{returns two alpha values $\alpha_1 \leq \alpha_2$, such as for
$\alpha$ between $\alpha_1$ and $\alpha_2$, the vertex is attached but singular, and
for $\alpha$ upper $\alpha_2$, the vertex is regular.}


\ccModifiers

\begin{ccAdvanced}

\ccMethod{void set_range(std::pair< FT, FT > I);}
{sets the alpha values $\alpha_1 \leq \alpha_2$, such as for
$\alpha$ between $\alpha_1$ and $\alpha_2$, the vertex is attached but singular, and
for $\alpha$ upper $\alpha_2$, the vertex is regular.}

\end{ccAdvanced}

\end{ccRefConcept}
