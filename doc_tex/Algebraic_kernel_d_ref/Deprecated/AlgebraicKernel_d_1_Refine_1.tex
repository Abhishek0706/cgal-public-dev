\begin{ccRefFunctionObjectConcept}{AlgebraicKernel_d_1::Refine_1}

\ccDefinition
An \ccc{AlgebraicKernel_d_1::AlgebraicReal_1} $x$ is bounded by some interval 
$I = [upper, lower]$, where $x \in I$. 
This functor refines (at least halves) the bounding interval of x.

\ccOperations
\ccCreationVariable{fo}
\ccThree{void++}{fo(first_argument_type,++}{}

A model \ccVar\ of this type must provide:

\ccMethod{
void operator()(AlgebraicKernel_d_1::AlgebraicReal_1& x);}{ 
  Refines (at least halves) the bounding interval of x. \\
  Note that an interval may also be degenerated to a single point.
} 

\ccMethod{
void operator()(AlgebraicKernel_d_1::AlgebraicReal_1& x, int rel_prec);}{ 
  This operator refines the current interval of $x$ with respect to
  the given relative precision. \\
  That is: $|lower - upper| / |x| \leq 2^{-rel\_prec}$, where $lower$
  and $upper$ are lower and upper bounds for $x$, respectively.\\
  In case $x$ equals zero the interval degenerates to the zero interval. 
}
